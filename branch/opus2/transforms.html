rule変換ルールの編集

// record: 1行分のオブジェクト。返り値で上書き const map = { 'Female':'女性', 'F':'女性', 'Male':'男性', 'M':'男性' }; const m = record.氏名.match(/（(.+?)）/); if (m) { record.性別 = m[1]; record.氏名 = record.氏名.replace(m[0], '').trim(); } record.性別 = map[record.gender] ?? record.性別; return record;
